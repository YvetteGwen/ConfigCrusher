program : stmt_block

  stmt_block : (stmt)+

stmt : variable EQUAL expr 
     | IF LEFT_PARENT expr RIGHT_PAREN { stmt_block } 
     | SLEEP LEFT_PARENT expr RIGHT_PARENT  

expr : term ((PLUS | MINUS | MULT | DIV) term)*

  term : variable 
     | INTEGER 
     | CONFIG 
     | NOT term  

variable : ('a'..'z')+

  CONFIG : ('A'..'Z')+
